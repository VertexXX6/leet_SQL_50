SELECT 
    ROUND(
        SUM(CASE WHEN order_date = customer_pref_delivery_date THEN 1 ELSE 0 END) * 100.0 / COUNT(*)
    , 2) AS immediate_percentage
FROM Delivery d
join (
    select customer_id , min(order_date) as 1st_order
    from Delivery 
    group by customer_id
)e
 on d.customer_id=e.customer_id
 and d.order_date=e.1st_order;
